def findDuplicate(nums: List[int]) -> int:    
    # loop 1
    slow1 = 0
    fast = 0
    while True:
        slow1 = nums[slow1]
        fast = nums[nums[fast]]
        if slow1 == fast:
            break

    # loop 2
    slow2 = 0
    while slow1 != slow2:
        slow1 = nums[slow1]
        slow2 = nums[slow2]
    return slow1
